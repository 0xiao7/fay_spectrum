GAS 後端設定
為什麼需要 GAS 中轉？

直接在前端呼叫 Notion API 會暴露你的 API Key（任何人打開 DevTools 都看得到）。 透過 GAS 當中轉層，前端只送查詢字串，Key 永遠在 GAS 伺服器端，安全多了。 同時 GAS 也能串接 LINE Bot 推播，打完高分自動通知你。

GAS Proxy 腳本（貼到 apps.script.google.com）
// ── Fay's Spectrum GAS Proxy ── // 在 GAS 指令碼屬性設定： // NOTION_API_KEY, NOTION_LAW_DB_ID, LINE_CHANNEL_ACCESS_TOKEN, LINE_USER_ID const P = PropertiesService.getScriptProperties(); function doGet(e) { const action = e.parameter.action; const q = e.parameter.q || ''; if (action === 'search') { return ContentService .createTextOutput(JSON.stringify(notionSearch(q))) .setMimeType(ContentService.MimeType.JSON); } if (action === 'star5') { return ContentService .createTextOutput(JSON.stringify(notionStar5())) .setMimeType(ContentService.MimeType.JSON); } if (action === 'push') { const msg = e.parameter.msg || ''; linePush(msg); return ContentService .createTextOutput('{"ok":true}') .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput('{"error":"unknown action"}'); } function notionSearch(q) { const key = P.getProperty('NOTION_API_KEY'); const db = P.getProperty('NOTION_LAW_DB_ID'); const isNum = /^\d+/.test(q); const filter = isNum ? {property:'條號', title:{contains:'§ '+q}} : {or:[ {property:'關鍵字', multi_select:{contains:q}}, {property:'條號', title:{contains:q}} ]}; const res = UrlFetchApp.fetch( 'https://api.notion.com/v1/databases/'+db+'/query', { method:'POST', headers:{Authorization:'Bearer '+key,'Notion-Version':'2022-06-28','Content-Type':'application/json'}, payload: JSON.stringify({filter, page_size:6}) } ); return JSON.parse(res.getContentText()); } function notionStar5() { const key = P.getProperty('NOTION_API_KEY'); const db = P.getProperty('NOTION_LAW_DB_ID'); const res = UrlFetchApp.fetch( 'https://api.notion.com/v1/databases/'+db+'/query', { method:'POST', headers:{Authorization:'Bearer '+key,'Notion-Version':'2022-06-28','Content-Type':'application/json'}, payload: JSON.stringify({ filter:{property:'重要性', select:{equals:'★★★★★'}}, page_size:10 }) } ); return JSON.parse(res.getContentText()); } function linePush(msg) { const token = P.getProperty('LINE_CHANNEL_ACCESS_TOKEN'); const uid = P.getProperty('LINE_USER_ID'); UrlFetchApp.fetch('https://api.line.me/v2/bot/message/push', { method:'post', headers:{Authorization:'Bearer '+token,'Content-Type':'application/json'}, payload: JSON.stringify({to:uid, messages:[{type:'text',text:msg}]}) }); }
